CMAKE_MINIMUM_REQUIRED(VERSION 2.8.4)
PROJECT(emas C)

set(EMAS_VERSION_MAJOR 1)
set(EMAS_VERSION_MINOR 0)
set(EMAS_VERSION_PATCH 0)
set(EMAS_VERSION ${EMAS_VERSION_MAJOR}.${EMAS_VERSION_MINOR}.${EMAS_VERSION_PATCH})

find_package(BISON REQUIRED)
find_package(FLEX REQUIRED)
find_package(emelf REQUIRED)

INCLUDE_DIRECTORIES(${CMAKE_SOURCE_DIR}/include)
INCLUDE_DIRECTORIES(${CMAKE_BINARY_DIR}/src)

ADD_DEFINITIONS(-Wall -std=c99 -D_XOPEN_SOURCE=700 -O2 -DEMAS_VERSION="${EMAS_VERSION}")

ADD_SUBDIRECTORY(src)
